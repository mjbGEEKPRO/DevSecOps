commande pour bloquer un port ou une ip

sudo ufw deny 22/tcp    # Bloque le port 22 TCP pour tout le monde
sudo ufw deny from 185.220.101.47  # Bloque cette IP sur tous les ports
sudo ufw deny from 185.220.101.47 to any port 22  # Bloque IP sur port 22 seulement
sudo ufw status verbose # Vérif


sudo iptables -A INPUT -s 185.220.101.47 -j DROP      # Bloque l’IP source
sudo iptables -A INPUT -p tcp --dport 22 -j DROP      # Bloque le port 22 TCP
sudo iptables -L -n -v # Vérif les règles
